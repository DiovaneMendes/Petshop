
CREATE TABLE cliente(
	id SERIAL PRIMARY KEY,
	nome VARCHAR(50) NOT NULL,
	rg INTEGER,
	telefone INTEGER
);

CREATE TABLE pet(
	id SERIAL PRIMARY KEY,
	nome VARCHAR(20) NOT NULL,
	tipo_animal VARCHAR(20) NOT NULL,
	cliente INTEGER REFERENCES CLIENTE(id),
	tipo_servico INTEGER REFERENCES TIPO_SERVICO(id)
);

CREATE TABLE tipo_servico(
        id SERIAL PRIMARY KEY,
	numero_servico INTEGER,
	nome VARCHAR(30) NOT NULL,
	tipo_atendimento VARCHAR(20) NOT NULL,
	preco DECIMAL
);

CREATE TABLE venda(
	id SERIAL PRIMARY KEY,
	cliente VARCHAR(50) REFERENCES CLIENTE(id),
	data_hora TIMESTAMP NOT NULL,
	tipo_servico INTEGER REFERENCES TIPO_SERVICO(id),
	valor_total DECIMAL 
);
